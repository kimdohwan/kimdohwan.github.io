list 로 구현하는 queue

```
# FIFO

def add(item):
    q.append(item)

def remove():
    if len(q) != 0:
        item = q.pop(0)
        return item

def print_q():
    print('front -> ', end='')
    for i in range(len(q)):
        # {!s:>8} 이거 문자열 포멧팅에서 좌우로 정렬하는 파이썬 문법
        # 파이썬 문서 string 에 설명 나온다
        print('{!s:>3}'.format(q[i]), end=' ') 
    print('<- rear')
```

singly linked list 로 구현하는 queue

```
class Node:
    def __init__(self, item, n):
        self.item = item
        self.next = n


def add(item):
    global size
    global front
    global rear
    new_node = Node(item, None)
    if size == 0:
        front = new_node
    else:
        rear.next = new_node
    rear = new_node
    size += 1

def remove():
    global size
    global front
    global rear
    if size != 0:
        fitem = front.item # return 해줄 front item 
        front = front.next # front 를 새로 지정(다음 항목
        size -= 1
        if size == 0: # size 가 1개였다가 -1되서 0이 된 경우 따로 정의
        # front = None, rear = None 정의, 
        # front 는 이미 front.next 값이 None 일것이기 때문에 또 정의 않해도 됨
            rear = None 
        return fitem

def print_q():
    p = front
    print('front: ', end='')
    while p:
        if p.next != None:
            print(p.item, '->   ', end='')
        else:
            print(p.item, end='')
        p = p.next
    print('   : rear')

front = None
rear = None
size = 0
```
